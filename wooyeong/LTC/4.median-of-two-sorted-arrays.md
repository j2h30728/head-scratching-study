function findMedianSortedArrays(nums1, nums2) {
    const arr = [...nums1, ...nums2].sort((a,b)=>a-b);

    if(arr.length %2 === 0){
        const idx =arr.length/2
        return  (arr[idx] + arr[idx - 1]) /2;;
    }else{
        const idx =Math.floor(arr.length/2);
        return arr[idx]
    }
};
