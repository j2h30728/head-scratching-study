function fib(n) {
    let memory = [0,1];
    function fib_dfs(n) {
        if(memory[n]!== undefined) return memory[n];

        memory[n] = fib_dfs(n - 1) + fib(n - 2);
        return memory[n];
    }
return fib_dfs(n)
}
