function solution(word) {
    const apv = ['A', 'E', 'I', 'O', 'U'];
    var answer =Number.MAX_SAFE_INTEGER;
    let cnt = 0;

    function dfs(stack){
        if(stack.join("") === word){
            answer =Math.min(answer, cnt)
            return ;
        }
        if(stack.length === 5){
            return;
        }

        for(let i =0; i< apv.length ;i++){
            stack.push(apv[i]);
            cnt++;
            dfs(stack);
            stack.pop();
        }

    }
    dfs([])
    return answer;
}
